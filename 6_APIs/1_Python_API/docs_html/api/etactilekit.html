 17class ETactileKit:
 18    """
 19    A class to manage and configure an E-Tactile Kit via an ESP32 microcontroller.
 20
 21    This class provides methods for sending parameter configurations (e.g., electrode count,
 22    stimulation pulse properties, sense pulse properties) and receiving voltage measurements
 23    from an ESP32. It relies on an external Communication object to handle the low-level
 24    serial communication.
 25
 26    Attributes:
 27        comm (Communication): The communication interface for sending/receiving data 
 28            to/from the ESP32.
 29        stimulation_polarity (int): Stimulation polarity mode (1 for Anodic, 0 for Cathodic).
 30        number_of_electrodes (int or None): The total number of electrodes in use.
 31        stimulation_pulse_height (int or None): The pulse height for stimulation.
 32        stimulation_pulse_width (int or None): The pulse width in µs for stimulation.
 33        sense_pulse_height (int or None): The pulse height for sensing.
 34        sense_pulse_width (int or None): The pulse width µs for sensing.
 35        channel_discharge_time (int or None): The time µs taken to discharge
 36            the channel after stimulation.
 37        stimulation_frequency (int or None): The frequency in Hz
 38            for stimulation pulses.
 39        hv513_num (int or None): The number of HV513 driver chips detected.
 40        electrode_mapping (list): Maps the logical electrode indices to hardware-specific 
 41            electrode indices.
 42        voltages (list): Stores the most recent voltage readings from each electrode.
 43    """

 46        """
 47        Initialize the ETactileKit object with a communication interface.

 49        Args:
 50            port_name (str): The name of the serial port (e.g., 'COM3' on Windows, '/dev/ttyUSB0' on Linux).
 51            baudrate (int): The communication speed in baud.
 52            timeout (float, optional): The default timeout for serial read operations. Defaults to 0.02.
 53        """
 54        self.comm = Communication(port_name, baudrate, timeout)  # Communication object for serial communication tasks
 55        #------------------------------------------------------------------------------------------#
 56        # Defined parameters for the ETactile Kit                                                  #
 57        #------------------------------------------------------------------------------------------#
 58        self.stimulation_polarity = 1  # 1 for Anodic, 0 for Cathodic
 59        self.number_of_electrodes = None

246    def send_stim_pattern(self, stim_pattern):
247        """
248        Send stimulation pattern to ESP32
249        
250        Args:
251            stim_pattern (list): A list of n integers(8bit representable - 0 < x <255) representing the intensity values of the n electrodes.
252        
253        Returns:
254            bool: True if the command was successfully acknowledged by the ESP32, False otherwise.
255        """
256        if len(stim_pattern) != self.number_of_electrodes:
257            print("Stimulation pattern length should be equal to the number of electrodes")
258            return False
259        
260        mapped_stim_pattern = [0]*self.number_of_electrodes
261        for ch in range(self.number_of_electrodes):
262            mapped_stim_pattern[self.electrode_mapping[ch]] = stim_pattern[ch]
263
264        self.comm.write_serial_bytes(PC_ESP32_STIM_PATTERN)
265
266        for ch in range(self.number_of_electrodes):
267            self.comm.write_serial_bytes(mapped_stim_pattern[ch])
268
269        rcv = self.comm.read_serial_bytes()
270        if rcv == ESP32_PC_RECEIVE_FINISHED:
271            return True
272        else:
273            return False

275    def get_voltage_readings(self):
276        """
277        Read the voltage data from the ESP32 corresponding to the number of electrodes defined in the application.
278        
279        Returns:
280            list: A list of voltage readings from each electrode.
281        """
282        self.comm.write_serial_bytes(PC_ESP32_MEASURE_REQUEST)
283        temp_voltages = []
284        for i in range(self.number_of_electrodes):
285            received_data = self.comm.read_serial_bytes()  # Read the voltage data and allocate more time since the reading is averaged over multiple readings internally
286            temp_voltages.append(received_data)
287
288        self.voltages = [temp_voltages[self.electrode_mapping[i]] for i in range(self.number_of_electrodes)]
289        return self.voltages

291    def update_and_get_hv513_count(self):
292        """
293        Get the count of HV513s connected. If only the main controller module is used this value will be 1. Each switching module adds 8 more HV513s
294        
295        Returns:
296            int: The number of HV513 driver chips connected to the ESP32.
297        """
298        self.comm.write_serial_bytes(PC_ESP32_HV513_NUM_REQUEST)
299        received_data = self.comm.read_serial_bytes_with_timeout()  # Read the count of HV513s and might need more time since the reading is averaged over multiple readings internally
300        self.hv513_num = received_data
301        return received_data

319    def check_valid_initialization(self):
320        """
321        Check the safety and the validity of the parameters set for the application.
322        The function will check whether the parameters set are safe or not.
323        
324        Returns:
325            bool: True if all parameters are set correctly, False otherwise.
326        """
327        if self.number_of_electrodes is None:
328            print("Number of electrodes is not set")
329            return False
330        if self.stimulation_pulse_height is None:
331            print("Stimulation pulse height is not set")
332            return False
333        if self.stimulation_pulse_width is None:
334            print("Stimulation pulse width is not set")
335            return False
336        if self.sense_pulse_height is None:
337            print("Sense pulse height is not set")
338            return False
339        if self.sense_pulse_width is None:
340            print("Sense pulse width is not set")
341            return False
342        if self.channel_discharge_time is None:
343            print("Channel discharge time is not set")
344            return False
345        if self.stimulation_frequency is None:
346            print("Stimulation frequency is not set")
347            return False
348        if self.hv513_num is None:
349            print("HV513 count is not set")
350            return False
351        if self.hv513_num*8 < self.number_of_electrodes:
352            print("Number of electrodes exceeds the number of outputs connected.\nPlease check the number of stacked switching modules")
353            return False
354        
355        print("All parameters are set")
356        return True

A class to manage and configure an E-Tactile Kit via an ESP32 microcontroller.

This class provides methods for sending parameter configurations (e.g., electrode count, stimulation pulse properties, sense pulse properties) and receiving voltage measurements from an ESP32. It relies on an external Communication object to handle the low-level serial communication.

Attributes: comm (Communication): The communication interface for sending/receiving data to/from the ESP32. stimulation_polarity (int): Stimulation polarity mode (1 for Anodic, 0 for Cathodic). number_of_electrodes (int or None): The total number of electrodes in use. stimulation_pulse_height (int or None): The pulse height for stimulation. stimulation_pulse_width (int or None): The pulse width in µs for stimulation. sense_pulse_height (int or None): The pulse height for sensing. sense_pulse_width (int or None): The pulse width µs for sensing. channel_discharge_time (int or None): The time µs taken to discharge the channel after stimulation. stimulation_frequency (int or None): The frequency in Hz for stimulation pulses. hv513_num (int or None): The number of HV513 driver chips detected. electrode_mapping (list): Maps the logical electrode indices to hardware-specific electrode indices. voltages (list): Stores the most recent voltage readings from each electrode.
